class ExerciseViewModel(private val repository: ExerciseRepository) : ViewModel() {
    val searchQuery: StateFlow<String>               // current search text
    val selectedMuscleGroup: StateFlow<String?>      // null = All
    val exercises: StateFlow<List<Exercise>>          // flat list (all filtered results)
    val groupedExercises: StateFlow<Map<String, List<Exercise>>>  // empty when search active
    val deleteResult: StateFlow<DeleteResult?>        // null = no pending result

    fun onSearchQueryChange(query: String)
    fun onMuscleGroupSelected(group: String?)        // null = All
    fun clearSearch()
    fun saveExercise(name: String, muscleGroup: String, equipmentType: String, existing: Exercise?)
    fun deleteExercise(exercise: Exercise)
    fun clearDeleteResult()
}

sealed class DeleteResult {
    object Deleted : DeleteResult()
    data class Blocked(val sessionCount: Int) : DeleteResult()
}

List layout rules:
- No search + no chip: grouped by muscle group, stickyHeader per group
- Search active: flat list, no headers regardless of chip state
- Chip active only: grouped, show only matching group header + items

Chip row: LazyRow below search bar, "All" chip first, then each muscle group

if (showBottomSheet) {
