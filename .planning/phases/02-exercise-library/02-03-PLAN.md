<!-- Research-specified ViewModel pattern -->
```kotlin
// combine() + flatMapLatest for dual StateFlow input → single reactive output
val exercises: StateFlow<List<Exercise>> = combine(_searchQuery, _selectedMuscleGroup) { q, mg ->
    Pair(q, mg)
}.flatMapLatest { (query, muscleGroup) ->
    repository.searchExercises(query, muscleGroup)
}.stateIn(viewModelScope, SharingStarted.WhileSubscribed(5_000), emptyList())

// Grouped map: non-empty only when search is blank
val groupedExercises: StateFlow<Map<String, List<Exercise>>> = exercises
    .combine(_searchQuery) { list, query ->
        if (query.isBlank()) list.groupBy { it.primaryMuscleGroup }
        else emptyMap()
    }
    .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5_000), emptyMap())
```
